titleSite = "Mon Site" //variables

element header monHead(){ //element de page head
    title : titleSite; //on peut utiliser des variables
    color : "#FFFFF";
    textcolor : "#FAFAFA";
}
element nav monNav(){
    color : "#FFFFF";
    text-color : "#FAFAFA";

    menu : [
       "home" : index,
       "gallerie" : gallery,
    ];
}

element footer monFoot(i){
    title : "mon foot"+i;
    color : "#FAFAFA";
    paragraph : "asfdasd";
    copyright : "(c) mon copyright";
}

//définition d’une page, chaque page est considérée comme une variable
page gallery{
    //paramètres obligatoires
    name : "ma galerie";
    address : "/gallery.htm";

        print monHead();
        print monNav();

        for(i=1;i<10;i++){
            print monFoot(i);
        }
}
